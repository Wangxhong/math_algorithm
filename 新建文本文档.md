 `public static String ConvertIntToBinary(int decimal)
	{
		if (decimal < 0)
		{
			int reverseNum = ((decimal * -1) ^ Integer.MAX_VALUE) + 1;
			return ConvertIntToBinary(reverseNum);
		}
		StringBuilder builder = new StringBuilder();
		while (decimal > 0)
		{
			if ((decimal & 1) != 0)
			{
				builder.insert(0, 1);
			}
			decimal = decimal >> 1;
		}
		return builder.toString();
	}`
